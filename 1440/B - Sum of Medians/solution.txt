t = int(input())
for _ in range(t):
    n, k = map(int,input().split())
    arr = list(map(int,input().split()))
    a = (n+1)//2 - 1 
    i = 0
    j = n*k-1
    som = 0
    for m in range(k):
        brr = arr[i:i+a] + arr[j-n+a+1:j+1]
        som+=brr[a]
        i+=a
        j-=(n-a)
    print(som)
